// Here's a simple function (not a closure), which we will use 
// as the closure's containing scope environment
fn sample() {

  let heavy_weight = 100;
  
  let is_heavy_closure = |weight: u32| { // A closure resembles a function, 
                                         // but with "||" rather than "()" around the parameters
    weight > heavy_weight  // Closures capture variables from their containing scope
  };
  
  println!("{:?}", is_heavy_closure(99)); // Closures are invoked like functions
                                          // prints false

  println!("{}", is_heavy_closure(101)); // prints true
  
  let is_heavy_closure_with_type_inference = |weight| { // Parameter types can be inferred
    weight > heavy_weight
  };
  
  let is_heavy_closure_without_braces = |weight| weight > heavy_weight; // If the return expression is a single line,
                                                                        // we don't need the curly braces

  return is_heavy_closure; // Error: can't be sent to another context
                           // ahaa
}
